var plusOne = function(digits) {
    const n = digits.length - 1;
    for (let i = n; i >= 0; --i) {
        if (digits[i] === 9)  digits[i] = 0;
        else {
            digits[i] += 1;
            break;
        }
        if (!i) digits.unshift(1);
    }
    return digits;
};
