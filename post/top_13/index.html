var romanToInt = function(s) {
    let res = 0;
    const table = new Map([['I', 1], ['V', 5], ['X', 10], ['L', 50], ['C', 100], ['D', 500], ['M', 1000]]);
    for (let i = 0; i < s.length - 1; ++i) {
        const m = table.get(s.charAt(i)), n = table.get(s.charAt(i + 1));
        res += m >= n ? m : -m; 
    }
    res += table.get(s.charAt(s.length - 1));
    return res;
};
