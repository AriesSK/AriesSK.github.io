    const n = matrix.length;
    for (let i = 0; i < n >> 1; ++i) {
        for (let j = i; j < n - i - 1; ++j) {
            const tmp = matrix[i][j];
            matrix[i][j] = matrix[n - j - 1][i];
            matrix[n - j - 1][i] = matrix[n - i - 1][n - j - 1];

            matrix[j][n - i - 1] = tmp;
        }
    }
    return matrix;

var rotate = function(matrix) {
    const n = matrix.length;
    for (let i = 0; i < n >> 1; ++i) {
        for (let j = 0; j < n; ++j) {
            const tmp = matrix[i][j];
            matrix[i][j] = matrix[n - i - 1][j];
            matrix[n - i - 1][j] = tmp;
        }
    }
    for (let i = 0; i < n; ++i) {
        for (let j = 0; j < i; ++j) {
            const tmp = matrix[i][j];

            matrix[j][i] = tmp;
        }
    }
