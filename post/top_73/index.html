var setZeroes = function(matrix) {
    const m = matrix.length, n = matrix[0].length;
    let x = false, y = false;

        if (!matrix[i][0]) {
            x = true;
            break;
        }
    }
    for (let j = 0; j < n; ++j) {
        if (!matrix[0][j]) {
            y = true;
            break;
        }
    }

        for (let j = 1; j < n; ++j) {
