var constructArr = function(a) {
    if (!a.length) return a;
    const b = new Array(a.length);
    b[0] = 1;
    for (let i = 1; i < a.length; ++i) b[i] = b[i - 1] * a[i - 1];
    let tmp = 1;
    for (let j = a.length - 2; j >= 0; --j) {
        tmp *= a[j + 1];
        b[j] *= tmp;
    }
    return b;
};
