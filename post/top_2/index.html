var addTwoNumbers = function(l1, l2) {
    let carry = 0, node = new ListNode();
    const res = node;
    while(l1 || l2) {
        const n1 = l1 ? l1.val : 0;
        const n2 = l2 ? l2.val : 0;

        node.next = new ListNode(sum % 10);
