var findContinuousSequence = function(target) {
    const res = [];
    for (let start = 1; start <= target >> 1; ++start) {
        let l = start, r = (target >> 1) + 1;
        while(l <= r) {
            const end = (l + r) >> 1;
            if (sum(start, end) === target) {
                const tmp = [];
                for (let i = start; i <= end; ++i) tmp.push(i);
                res.push(tmp);
                break;
            }
            if (sum(start, end) < target) l = end + 1;
            else r = end - 1;
        }
    }
    return res;
};
