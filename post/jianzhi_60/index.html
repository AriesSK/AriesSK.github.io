var dicesProbability = function(n) {
    const dp = new Array(n);
    dp[0] = new Array(6).fill(1 / 6);
    for (let i = 1; i < n; ++i) dp[i] = new Array((i + 1) * 5 + 1).fill(0);
    for (let i = 1; i < n; ++i) {
        for (let j = 0; j < dp[i - 1].length; ++j) {
            for (let k = 0; k < 6; ++k) {
                dp[i][j + k] += dp[i - 1][j] / 6;
            }
        }
    }
    return dp[n - 1];
};
